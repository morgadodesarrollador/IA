apiVersion: apps/v1
kind: Deployment
metadata:
  name: ollama
spec:
  replicas: {{ .Values.replicaCount }}
  selector:
    matchLabels:
      app: ollama
  template:
    metadata:
      labels:
        app: ollama
    spec:

      initContainers:
        - name: pull-models
          image: {{ .Values.image.repository }}:{{ .Values.image.tag }}
          command: ["sh", "-c"]
          args:
            - |
              ollama serve &
              sleep 5
              {{- range .Values.model.list }}
              ollama pull {{ . }}
              {{- end }}
              killall ollama
          volumeMounts:
            - name: ollama-data
              mountPath: /root/.ollama

      containers:
        - name: ollama
          image: {{ .Values.image.repository }}:{{ .Values.image.tag }}
          ports:
            - containerPort: 11434
          volumeMounts:
            - name: ollama-data
              mountPath: /root/.ollama
          resources:
            {{- toYaml .Values.resources | nindent 12 }}

      volumes:
        - name: ollama-data
          persistentVolumeClaim:
            claimName: ollama-pvc
